php console/yii migrate/create create_table_game_gamecategory
//执行一个迁移
php console/yii

//创建一个rbac迁移,会在common/migrations/rbac目录生成一个新的迁移文件
php console/yii rbac-migrate/create
//执行一个rbac迁移文件
php console/yii rbac-migrate
